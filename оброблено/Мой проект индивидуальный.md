## обединения трех таблиц 
- **`users`** — таблица с информацией о пользователях.
- **`roles`** — таблица с информацией о ролях.
- **`user_roles`** — таблица, которая связывает пользователей и их роли.
- **`short_url`** — таблица с сокращенными URL-адресами.
SELECT 
    u.id AS user_id,
    u.user_name AS user_name,
    u.email AS user_email,
    r.id AS role_id,
    r.name AS role_name,
    s.id AS short_url_id,
    s.short_key AS short_key,
    s.full_url AS full_url,
    s.click_count AS click_count,
    s.creation_date AS creation_date
FROM 
    users u
JOIN 
    user_roles ur ON u.id = ur.user_id
JOIN 
    roles r ON ur.role_id = r.id
JOIN 
    short_url s ON u.id = s.user_id;
